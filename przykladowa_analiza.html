# Translate all timestamps to milliseconds
import dateutil
import datetime

def convert_psychopy_timestamp(timestamp_string):
    dt = dateutil.parser.parse(timestamp_string)  # curr time
    epoch = datetime.datetime.utcfromtimestamp(0) # timestamp 0
    ts = (dt - epoch).total_seconds() * 1000.0    # get time difference in milliseconds
    return ts

# PsychoPy timestamps ('2017-04-26 13:26:19.479000')
for dataset in [ob1images, ob2images, ob3images, ob4images, ob5images, ob6images,
                ob11images, ob12images, ob13images, ob14images, ob15images,
                ob16images, ob17images, ob18images, ob19images]:
    dataset.Timestamp = dataset.Timestamp.apply(func=convert_psychopy_timestamp)

    dataset.Timestamp = dataset.Timestamp.apply(func=lambda t: 1000.0 * (t + 2*60*60))

# generate the mean time-series for each group

def fill_missing_values(data):
    for col in range(data.shape[1]):
        col_vals = data.ix[:,col]
        last_val = 0
        nan_count = 0
        for val in range(len(col_vals)):
            if pandas.isnull(col_vals[val]) or (abs(col_vals[val]) < 0.0000001):
                nan_count += 1
                continue
            if nan_count > 0:
                diff_val = (col_vals[val] - last_val) / (nan_count+1)
                for idx in range(1,nan_count+1):
                    col_vals[val-idx] = col_vals[val] - idx*diff_val
                nan_count = 0
            last_val = col_vals[val]
        data.ix[:,col] = col_vals
    return data


def calculate_mean(dataset):
    # resample the data that for each second ('s') there is only one probe (mean of the probes)
    result = dataset.resample('s').mean()
    
    # if there are seconds without any probe, fill it with the mean value of neighbours
    result = fill_missing_values(result)
    
    # calculate the mean value time-series across all stimulus within frame
    return result.mean(axis=1)


neutral_hr_MEAN = calculate_mean(neutral_hr)
